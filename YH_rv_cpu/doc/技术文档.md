## 1. 系统架构

### 1.1 整体架构

YH_rv_cpu 是一个参数化的 RISC-V 处理器核，支持 XLEN=32 和 XLEN=64 两种数据通路宽度。处理器采用五级流水线结构，分别为：

- **IF (Instruction Fetch)**: 取指级
- **ID (Instruction Decode)**: 译码级
- **EX (Execute)**: 执行级
- **MEM (Memory Access)**: 访存级
- **WB (Write Back)**: 写回级

### 1.2 顶层模块接口

```verilog
module YH_rv_cpu #(
    parameter integer XLEN = 32,
    parameter integer IMEM_SYNC = 0,
    parameter integer IMEM_OUTPUT_REG = 0,
    parameter integer DMEM_SYNC = 0,
    parameter [XLEN-1:0] RESET_VECTOR = {XLEN{1'b0}}
) (
    input  wire            clk,
    input  wire            rst_n,
    input  wire            timer_irq,
    output wire            imem_req,
    output wire [XLEN-1:0] imem_addr,
    input  wire [31:0]     imem_rdata,
    input  wire            imem_rvalid,
    output wire [XLEN-1:0] dmem_addr,
    input  wire [XLEN-1:0] dmem_rdata,
    input  wire            dmem_rvalid,
    output wire            dmem_read_req,
    output wire [XLEN-1:0] dmem_wdata,
    output wire [XLEN/8-1:0] dmem_wstrb,
    output wire            trap,
    output wire [XLEN-1:0] debug_pc
);
```

### 1.3 参数化设计

| 参数 | 默认值 | 说明 |
|------|--------|------|
| XLEN | 32 | 数据通路宽度，支持 32/64 |
| IMEM_SYNC | 0 | 指令存储器同步模式 |
| IMEM_OUTPUT_REG | 0 | 指令存储器输出寄存器 |
| DMEM_SYNC | 0 | 数据存储器同步模式 |
| RESET_VECTOR | 0x0 | 复位向量地址 |

### 2.2 流水线控制优先级

```verilog
// 控制优先级从高到低
assign csr_mstatus_trap_write = pipeline_run && ex_trap_valid;
assign csr_mstatus_mret_write = pipeline_run && !ex_trap_valid && ex_mret_valid;
```

**优先级顺序：**
1. `rst_n` - 异步复位
2. `trap_r` - 全局 trap 状态
3. `ex_trap_valid` - 同步异常/中断
4. `ex_mret_valid` - MRET 指令
5. `ex_redirect_valid` - 分支/跳转重定向
6. `stall_decode` - Load-Use 冒险暂停

### 2.3 流水线数据通路

### 4.3 分支判断逻辑

```verilog
assign branch_taken =
    branch && (
        ((branch_funct3 == 3'b000) &&  alu_eq)  ||  // BEQ
        ((branch_funct3 == 3'b001) && !alu_eq)  ||  // BNE
        ((branch_funct3 == 3'b100) &&  alu_lt)  ||  // BLT
        ((branch_funct3 == 3'b101) && !alu_lt)  ||  // BGE
        ((branch_funct3 == 3'b110) &&  alu_ltu) ||  // BLTU
        ((branch_funct3 == 3'b111) && !alu_ltu)    // BGEU
    );
```

### 4.4 JALR 目标计算

```verilog
assign rs1_plus_imm = rs1_value + imm;
assign jalr_target = {rs1_plus_imm[XLEN-1:1], 1'b0};  // 位0清零
```

### 4.5 访存地址对齐检测

```verilog
assign misaligned_mem =
    (load || store) && (
        ((mem_size == `YH_rv_cpu_MEM_H) && mem_addr[0]) ||
        ((mem_size == `YH_rv_cpu_MEM_W) && (mem_addr[1:0] != 2'b00)) ||
        ((mem_size == `YH_rv_cpu_MEM_D) && (mem_addr[2:0] != 3'b000))
    );
```

### 4.6 Store 数据格式化

```verilog
always @* begin
    case (mem_size)
        `YH_rv_cpu_MEM_B: begin
            store_data = {{(XLEN-8){1'b0}}, rs2_value[7:0]} << {byte_offset, 3'b000};
            store_wstrb = {{(STRB_W-1){1'b0}}, 1'b1} << byte_offset;
        end
        `YH_rv_cpu_MEM_H: begin
            store_data = {{(XLEN-16){1'b0}}, rs2_value[15:0]} << {byte_offset, 3'b000};
            store_wstrb = {{(STRB_W-2){1'b0}}, 2'b11} << byte_offset;
        end
        `YH_rv_cpu_MEM_W: begin
            store_data = {{(XLEN-32){1'b0}}, rs2_value[31:0]} << {byte_offset, 3'b000};
            store_wstrb = {{(STRB_W-4){1'b0}}, 4'hf} << byte_offset;
        end
        default: begin
            store_data = rs2_value;
            store_wstrb = {STRB_W{1'b1}};
        end
    endcase
end
```

### 5.2 Load 数据重组

```verilog
assign shifted_rdata = dmem_rdata >> {byte_offset, 3'b000};

always @* begin
    case (mem_size)
        `YH_rv_cpu_MEM_B: begin
            load_data = mem_unsigned ? {{(XLEN-8){1'b0}}, shifted_rdata[7:0]}
                                            : {{(XLEN-8){shifted_rdata[7]}}, shifted_rdata[7:0]};
        end
        `YH_rv_cpu_MEM_H: begin
            load_data = mem_unsigned ? {{(XLEN-16){1'b0}}, shifted_rdata[15:0]}
                                             : {{(XLEN-16){shifted_rdata[15]}}, shifted_rdata[15:0]};
        end
        `YH_rv_cpu_MEM_W: begin
            load_data = mem_unsigned ? {{(XLEN-32){1'b0}}, shifted_rdata[31:0]}
                                             : {{(XLEN-32){shifted_rdata[31]}}, shifted_rdata[31:0]};
        end
        default: begin
            load_data = shifted_rdata;  // LD / 无符号扩展场景
        end
    endcase
end
```

### 5.3 同步/异步数据访问

```verilog
assign dmem_read_req = valid && load;
```

### 7.2 写回旁路

```verilog
assign rs1_rdata =

    regs[rs1_addr];

assign rs2_rdata =
    (rs2_addr == 5'd0) ? {XLEN{1'b0}} :
    (rd_wen && (rd_addr == rs2_addr) && (rd_addr != 5'd0)) ? rd_wdata :
    regs[rs2_addr];
```

**关键特性：**
- x0 寄存器硬编码为 0
- 写回旁路：同周期写回的数据能被译码级直接看到
- 双端口异步读取，单端口同步写入

### 8.2 Load-Use 冒险检测

```verilog
assign load_use_hazard =
    id_ex_valid && id_ex_load && id_ex_rd_en && (id_ex_rd_addr != 5'd0) &&
    (
        (if_id_rs1_en && (if_id_rs1_addr == id_ex_rd_addr)) ||
        (if_id_rs2_en && (if_id_rs2_addr == id_ex_rd_addr))
    );

assign stall_decode = load_use_hazard || ex_mem_load_hazard;
```

**处理策略：** 插入 bubble 暂停流水线

### 8.3 前递路径

```verilog
always @* begin
    forward_a_sel = 2'b00;  // 来自寄存器堆
    forward_b_sel = 2'b00;  // 来自寄存器堆

    // EX/MEM -> ID 前递 (非 Load)
    if (id_ex_rs1_en && ex_mem_valid && ex_mem_rd_en && !ex_mem_load &&
        (ex_mem_rd_addr != 5'd0) && (ex_mem_rd_addr == id_ex_rs1_addr)) begin
        forward_a_sel = 2'b01;  // 来自 EX/MEM 流水线寄存器
    end

    // MEM/WB -> ID 前递
    if (id_ex_rs1_en && mem_wb_valid && mem_wb_rd_en &&
        (mem_wb_rd_addr != 5'd0) && (mem_wb_rd_addr == id_ex_rs1_addr)) begin
        forward_a_sel = 2'b10;  // 来自 MEM/WB 流水线寄存器
    end
    // ... 对 rs2 类似处理
end
```

**前递选择编码：**
| Sel | 数据来源 |
|-----|----------|
| 2'b00 | 寄存器堆 |
| 2'b01 | EX/MEM 流水线寄存器 (ALU结果或PC+4) |
| 2'b10 | MEM/WB 流水线寄存器 (ALU结果/Load数据/PC+4) |

```verilog
`define YH_rv_cpu_TRAP_ILLEGAL_INSN      32'd2
`define YH_rv_cpu_TRAP_BREAKPOINT        32'd3
`define YH_rv_cpu_TRAP_LOAD_MISALIGNED   32'd4
```

### 9.3 CSR 读写操作

```verilog
// CSR 命令
`define YH_rv_cpu_CSR_RW    2'd0  // 写
`define YH_rv_cpu_CSR_RS    2'd1  // 读-置位
`define YH_rv_cpu_CSR_RC    2'd2  // 读-清除

assign csr_write_data_ex =
    (id_ex_csr_cmd_r == `YH_rv_cpu_CSR_RW) ? csr_write_operand_ex :
    (id_ex_csr_cmd_r == `YH_rv_cpu_CSR_RS) ? (csr_rdata_ex | csr_write_operand_ex) :
    (csr_rdata_ex & ~csr_write_operand_ex);
```

### 9.4 Trap 检测逻辑

```verilog
assign ex_trap_cause =
    ex_interrupt_valid ? {{(XLEN-32){1'b0}}, `YH_rv_cpu_TRAP_MTIME_INTERRUPT} :
    (id_ex_ecall_r) ? {{(XLEN-32){1'b0}}, `YH_rv_cpu_TRAP_ECALL_MMODE} :
    (id_ex_ebreak_r) ? {{(XLEN-32){1'b0}}, `YH_rv_cpu_TRAP_BREAKPOINT} :
    (!csr_read_valid_ex && id_ex_csr_valid_r) ? {{(XLEN-32){1'b0}}, `YH_rv_cpu_TRAP_ILLEGAL_INSN} :
    (id_ex_illegal_r) ? {{(XLEN-32){1'b0}}, `YH_rv_cpu_TRAP_ILLEGAL_INSN} :
    (id_ex_load_r) ? {{(XLEN-32){1'b0}}, `YH_rv_cpu_TRAP_LOAD_MISALIGNED} :
    {{(XLEN-32){1'b0}}, `YH_rv_cpu_TRAP_STORE_MISALIGNED};

// 中断检测
assign ex_interrupt_valid =
    id_ex_valid_r &&
    !ex_mem_misaligned &&
    !id_ex_ecall_r &&
    !id_ex_ebreak_r &&
    !id_ex_mret_r &&
    !id_ex_illegal_r &&
    !csr_access_illegal_ex &&
    (csr_mstatus_r & `YH_rv_cpu_MSTATUS_MIE) != ZERO_XLEN &&
    (csr_mie_r & `YH_rv_cpu_MIE_MTIE) != ZERO_XLEN &&
    timer_irq;
```

```verilog
always @(posedge clk or negedge rst_n) begin
    if (!rst_n) begin
        timer_value_r  <= 64'h0000_0000_0000_0000;
        timer_cmp_r    <= 64'hffff_ffff_ffff_ffff;

        end
    end
end
```

## 13. 固件与启动流程

### 13.1 启动代码 (crt0.S)

```assembly
    .section .text.init
    .globl _start

_start:
    la sp, _stack_top    ; 初始化栈指针
    call main           ; 调用主函数

1:
    j 1b                ; 原地循环 (正常不应到达)
```

### 13.2 Trap 入口 (trap_entry.S)

```assembly
    .section .text.trap
    .globl toylog_trap_handler
    .align 2

toylog_trap_handler:
    addi sp, sp, -16     ; 保存上下文
    sw t0, 0(sp)
    sw t1, 4(sp)
    sw t2, 8(sp)

    csrr t0, mcause      ; 读取异常原因
    li t1, 11            ; ECALL 异常码
    bne t0, t1, trap_fail

    li t0, 0x10000000    ; UART 地址
    li t1, 't'           ; 输出 't'
    sw t1, 0(t0)

    csrr t2, mepc        ; 读取返回地址
    addi t2, t2, 4       ; 指向下一条指令
    csrw mepc, t2

    lw t0, 0(sp)         ; 恢复上下文
    lw t1, 4(sp)
    lw t2, 8(sp)
    addi sp, sp, 16
    mret                  ; 返回
```

### 13.3 链接脚本 (YH_rv_cpu.ld)

```ld
MEMORY
{
    ROM (rx)  : ORIGIN = 0x00000000, LENGTH = 16K
    RAM (rwx) : ORIGIN = 0x00004000, LENGTH = 16K
}

{
    .text : {
        *(.text.init)
        *(.text*)
        *(.rodata*)
    } > ROM

    .data : {
        *(.data*)
    } > RAM AT > ROM

    .bss : {
        *(.bss*)
        *(COMMON)
    } > RAM
```
